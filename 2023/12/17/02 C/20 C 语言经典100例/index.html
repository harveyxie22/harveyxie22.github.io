经典例子 001：无重复三位数

题目：有 1、2、3、4 四个数字，能组成多少个互不相同且无重复数字的三位数？都是多少？

解：

三位数字分别在个位、十位和百位，1、2、3、4 四个数字均可出现在个、十和百位。

#include <stdio.h>

int main(void)
{
    int num1, num2, num3; // 个位、十位、百位
    for (int i = 1; i <= 4; i++)
    {
        for (int j = 1; j <= 4; j++)
        {
            for (int k = 1; k <= 4; k++)
            {
                if (i != j && j != k && i != k)
                {
                    num1 = i;
                    num2 = j;
                    num3 = k;
                    printf("%d,%d,%d\n", num3, num2, num1);
                }
            }
        }
    }
}

经典例子 002：企业发奖金

题目：企业发放的奖金根据利润提成。

  • 利润(I)低于或等于10万元时，奖金可提10%；
  • 利润高于10万元，低于20万元时，低于10万元的部分按10%提成，高于10万元的部分，可提成7.5%；
  • 20万到40万之间时，高于20万元的部分，可提成5%；
  • 40万到60万之间时，高于40万元的部分，可提成3%；
  • 60万到100万之间时，高于60万元的部分，可提成1.5%；
  • 高于100万元时，超过100万元的部分按1%提成。

从键盘输入当月利润I，求应发放奖金总数？

解：

#include <stdio.h>
int main(void)
{
    int l;
    scanf("%d", &l);
    printf("Your input:%d\n", l);

    double b1, b2, b4, b6, b10, b;
    b1 = 100000 * 0.1;
    b2 = b1 + 100000 * 0.075;
    b4 = b2 + 200000 * 0.05;
    b6 = b4 + 200000 * 0.03;
    b10 = b6 + 400000 * 0.015;

    if (l <= 100000)
    {
        b = l * 0.1;
    }
    else if (l <= 200000)
    {
        b = b1 + (l - 100000) * 0.075;
    }
    else if (l <= 400000)
    {
        b = b2 + (l - 200000) * 0.05;
    }
    else if (l <= 600000)
    {
        b = b4 + (l - 400000) * 0.03;
    }
    else if (l <= 1000000)
    {
        b = b6 + (l - 600000) * 0.015;
    }
    else if (l >= 1000000)
    {
        b = b10 + (l - 1000000) * 0.01;
    }

    printf("bonus:%.2lf\n", b);
}

经典例子 003：完全平方数

题目：一个整数，它加上100后是一个完全平方数，再加上168又是一个完全平方数，请问该数是多少？

解：

#include <stdio.h>
#include <math.h>

_Bool isPerfectSquare(int num)
{
    int result = sqrt(num);
    return (result * result) == num ? 1 : 0;
}

int main(void)
{
    for (int i = 0; i < 10000; i++)
    {
        if (isPerfectSquare(i + 100) && isPerfectSquare(i + 100 + 168))
        {
            printf("number:%d\n", i);
        }
    }
}

最佳答案：

假设该数为 x，则：

\begin{align} x + 100 = n^2 \qquad &\text{(1)} \\ x + 100 + 168 = m^2 \qquad & \text{(2)} \\ \end{align}

由 (1) 和 (2) 可得：

m2−n2=(m+n)(m−n)=168(3)m^2 - n^2 = (m+n)(m-n) = 168 \qquad \text{(3)} \\ m2−n2=(m+n)(m−n)=168(3)

设 i=m+ni=m+ni=m+n , j=m−nj=m-nj=m−n，公式 (3) 的偶数性质，iii 或 jjj 中至少有一个为偶数。此时：

\begin{align} m = \frac{i+j}{2} \qquad &\text{(4)} \\ n = \frac{i-j}{2} \qquad &\text{(5)} \\ i * j = 168 \qquad &\text{(6)} \\ \end{align}

因为 mmm 和 nnn 都是整数，根据公式 (4) 可以推断：iii 和 jjj 要么都是奇数，要么都是偶数。但是根据前面判断它们中至少有一个为偶数，所以 iii 和 jjj 一定都是偶数，且 i≥2,j≥2,i>ji \geq 2, j \geq 2, i > ji≥2,j≥2,i>j。

根据公式 (6)，

1<i<1682+1(7)1 < i < \frac{168}{2} + 1 \qquad \text{(7)} \\ 1<i<2168​+1(7)

因此只需要遍历寻找 iii 的值，且同时满足 (4)(5)(7) 即可。以下是代码实现：

#include <stdio.h>
#include <math.h>

int main()
{
    int i, j, m, n, x;
    for (int i = 1; i < 168 / 2 + 1; i++)
    {
        if (168 % i == 0)
        {
            j = 168 / i;
            if (i > j && (i + j) % 2 == 0 && (i - j) % 2 == 0)
            {
                m = (i + j) / 2;
                n = (i - j) / 2;
                x = n * n - 100;
                printf("%d + 100 = %d * %d \n", x, n, n);
                printf("%d + 268 = %d * %d \n", x, m, m);
                printf("The number is %d\n", x);
            }
        }
    }
}

经典例子 004：判断一年中的第几天

题目：输入某年某月某日，判断这一天是这一年的第几天？

解：

#include <stdio.h>

int main()
{
    int year, month, day;
    printf("Please input the year:\n");
    scanf("%d", &year);
    printf("Please input the month:\n");
    scanf("%d", &month);
    printf("Please input the day:\n");
    scanf("%d", &day);
    printf("Your input:%d/%d/%d", year, month, day);

    if (month - 1 < 0 || day < 0)
    {
        printf("Your input:%d/%d/%d is incorrect.\n", year, month, day);
        return 0;
    }

    int sum = 0;
    switch (month - 1)
    {
    case 1:
        sum += 31;
        break;
    case 2:
        sum += 59;
        break;
    case 3:
        sum += 90;
        break;
    case 4:
        sum += 120;
        break;
    case 5:
        sum += 151;
        break;
    case 6:
        sum += 181;
    case 7:
        sum += 212;
        break;
    case 8:
        sum += 243;
        break;
    case 9:
        sum += 273;
        break;
    case 10:
        sum += 304;
        break;
    case 11:
        sum += 334;
        break;
    case 12:
        sum += 365;
        break;
    default:
        break;
    }

    if (year % 4 == 0 && year % 100 != 0 || year % 400 == 0)
    {
        sum += 1;
    }

    sum += day;
    printf("The number is %d day of the year.", sum);
}

经典例子 005：三个数由小到大输出

题目：输入三个整数x,y,z，请把这三个数由小到大输出。

解：

#include <stdio.h>

int main()
{
    int x = 0, y = 0, z = 0;
    printf("Please input three numbers:");
    scanf("%d %d %d", &x, &y, &z);

    if (x > y)
    {
        int temp = y;
        y = x;
        x = temp;
    }

    if (x > z)
    {
        int temp = x;
        x = z;
        z = temp;
    }

    if (y > z)
    {
        int temp = y;
        y = z;
        z = temp;
    }

    printf("%d,%d,%d", x, y, z);
}

经典例子 007：输出9*9口诀。

#include <stdio.h>

int main(void)
{
    for (int i = 1; i <= 9; i++)
    {
        for (int j = 1; j <= i; j++)
        {
            printf("%d * %d = %d\t", i, j, i * j);
        }
        printf("\n");
    }
}
